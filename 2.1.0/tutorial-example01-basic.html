            var dom = tui.dom;
        var container = dom.find('#fl');

        var layer = new tui.FloatingLayer(container);
        layer.setContent('<div id="modal">modal layer<button>&times;</button></div>');
        layer.setBound({top: 50, left: 50, width: 200, height: 150});

        var layer2 = new tui.FloatingLayer(container, {modaless: true});
        layer2.setContent('<div id="modaless">modaless layer<button>&times;</button></div>');
        layer2.setBound({top: 100, left: 100, width: 300, height: 200});

        dom.on(layer.container, 'click', function(e) {
            var target = e.target || e.srcElement;

            if (target.nodeName.toLowerCase() === 'button') {
                layer.hide();
            }
        });

        dom.on(dom.find('#open1'), 'click', function() {
            layer.show();
        });

        dom.on(layer2.container, 'click', function(e) {
            var target = e.target || e.srcElement;

            if (target.nodeName.toLowerCase() === 'button') {
                layer2.hide();
            }
        });

        dom.on(dom.find('#open2'), 'click', function() {
            layer2.show();
        });
        
                <div id="fl"></div>
            <button type="button" id="open1">Open layer1 (modal)</button>
            <button type="button" id="open2">Open layer2</button>
